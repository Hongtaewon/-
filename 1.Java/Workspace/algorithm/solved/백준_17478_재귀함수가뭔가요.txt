import java.io.*;
import java.util.*;

class Main {
	
	static int n;
	public static void main(String[] args) throws Exception {
		BufferedReader br = new BufferedReader(new InputStreamReader(System.in));
		n = Integer.parseInt(br.readLine());
		System.out.println("어느 한 컴퓨터공학과 학생이 유명한 교수님을 찾아가 물었다.");
		
		
		recur(0);
	}

	public static void recur(int i) {
		String[] st1 = new String[5];
		st1[0] = "\"재귀함수가 뭔가요?\"";
		st1[1] = "\"잘 들어보게. 옛날옛날 한 산 꼭대기에 이세상 모든 지식을 통달한 선인이 있었어.";
		st1[2] = "마을 사람들은 모두 그 선인에게 수많은 질문을 했고, 모두 지혜롭게 대답해 주었지.";
		st1[3] = "그의 답은 대부분 옳았다고 하네. 그런데 어느 날, 그 선인에게 한 선비가 찾아와서 물었어.\"";
		st1[4] = "라고 답변하였지.";

		String time = "";
		for(int j = 0;j<i;j++)
		{
			time = time + "____";
		}
		
		
		for(int k = 0;k<4;k++)
		{
			
			if(i == n)
			{
				System.out.println(time+st1[0]);
				System.out.println(time+"\"재귀함수는 자기 자신을 호출하는 함수라네\"");
				break;
			}
			else
			{	
				System.out.println(time+st1[k]);
			}
		}
		if(i != n)
			recur(i+1);
		System.out.println(time+st1[4]);
	}
}
